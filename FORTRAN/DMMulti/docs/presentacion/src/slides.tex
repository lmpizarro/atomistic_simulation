        \item<+->Interacciones de a pares a través de un potencial de Lenard-Jones
        \begin{block}<+->{Reglas de combinación}
		      \vspace{-1em}
          \begin{align*}
		        \epsilon_{ij} &= \sqrt{\epsilon_i \epsilon_j} \qquad \text{(Berthelot)} \\
            \sigma_{ij}   &= \frac{\sigma_i + \sigma_j}{2} \qquad \text{(Lorentz)}
          \end{align*}
	      \end{block}
		    \item<+-> Radio de corte:
		    \begin{equation*}
		      r^c_{ij} = 2.5 \sigma_{ij}
		    \end{equation*}
	      \item<+-> Ensamble ($N$,$V$,$T$)
        \item<+-> Se calcula el parámetro de red $a$ adecuado

\begin{frame}{Parámetros utilizados}
	\begin{itemize}
	\item<+-> La configuración inicial siempre fue un cristal FCC
	\item<+-> Se buscó que el cristal mantuviera la estructura 
	\begin{block}<+->{Parámetros de las pariculas}
	\begin{table}
	\begin{tabular}{ccccc}
		\toprule
		 Partícula & $\epsilon^*$ & $\sigma^*$ & $m^*$ \\ \midrule
		$1$ & $1.0$  & $1.0$ & $1.0$  \\ 
		$2$ & $0.8$ & $0.9$ & $0.6$  \\
		\bottomrule
	\end{tabular}
	\end{table}
	\end{block}
	\item<+->Las magnitudes reducidas están normalizadas con los parámetros de L-J de la partícula $1$
	\item<+->Estructura cristalina muy dependiente de los valores de $\sigma^*$
	\end{itemize}
\end{frame}

\begin{frame}{Determinación del parámetro de red}
  \begin{columns}
    \begin{column}{0.45\linewidth}
      \begin{itemize}
        \item<+->Se busca encontrar el valor de a que minimice la energía libre de Helmholtz 
	\begin{equation*}
	  F = U - T S 
 	 \end{equation*}
	\item<+-> Se simula a $T=0$ variando el parámetro de red hasta encontrar un mínimo en $U$
	\item<+-> Se asume que el valor encontrado será válido para $T\neq0$.
      \end{itemize}
    \end{column}

\section{Vibraciones}

\begin{frame}{Transformada de Fourier}
	\begin{itemize}
		\item Representación de una función como suma infinita de funciones senoidales 
		\item Pasamos al dominio de la frecuencia
		\item En el campo digital aplicamos Transformada rápida de fourier
		\item Dos posibilidades fftw3 o numpy
		\item Trabajamos con las 2, más flexible numpy
	\end{itemize}
\end{frame}

\begin{frame}{Análisis espectral de la velocidad}
	\begin{itemize}
            \item  Elegimos 2 configuraciones con relaciones de masa bien distintas
            \item  Relaciones de masas: $1.2$:$1.0$ y  $8.0$:$1.0$
	    \item  Acumulamos velocidades para los 4 átomos de una celda fcc	    
	\end{itemize}
\end{frame}

\begin{frame}{Intensidad media  espectral}
	\begin{itemize}
		\item O valor RMS espectral de la señal
		\item Sumar todos los componentes de la PSD y dividir por las M frecuencias
		\item Gráficos en función de las N corridas
	\end{itemize}
\end{frame}

\begin{frame}{Intensidad media: resultados }
	\begin{itemize}
		\item La intensidad media espectral de las partículas disminuye
			con la masa de la partícula.
		\item La relación de intensidades medias de los 2 sistemas
			es la relación de masas del sistema.
		\item La intensidad media de la partícula con mayor intensidad
			en el sistema pesado es mayor que en la del sistema liviano.
		\item Hay una disminución de la intensidad espectral con el aumento
			de las masas que intervienen en el sistema.
	\end{itemize}


\end{frame}

\section{Procedimientos}

\begin{frame}{Reactímetro}
	\begin{block}{Ecuaciones de la cinética puntual}
	\begin{align*}
		\frac{dC_i(t)}{dt} &= \frac{b_i}{\Lambda^*} n(t) - \lambda_i C_i(t) \hspace{1cm} i=1,...,15  \\
		\$(t) &= 1 + \frac{\Lambda^*}{n(t)} \left[\frac{dn(t)}{dt} - \sum_{i=1}^{15} \lambda_i C_i(t)\right] 
	\end{align*}
	\end{block}
    \begin{itemize}
        \item $n(t)$ proporcional a la tasa de detección
        \item Seis grupos de neutrones retardados (Tuttle)
        \item Nueve grupos de fotoneutrones (Keepin)
        \item Discretización asumiendo evolución lineal de $n(t)$ en un paso temporal. Integración analítica de las ecuaciones de los precursores
        \item Paso de discretización de $0.1\,s$
    \end{itemize}
\end{frame}
